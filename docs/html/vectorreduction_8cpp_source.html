80static __global__ void block_reduce(const double* input,
81 double* output,
82 size_t size)
83{
84 __shared__ double sdata[THREADS_PER_BLOCK];
85
86 const unsigned int tid = threadIdx.x;
87 const size_t idx = static_cast<size_t>(blockIdx.x) * blockDim.x * 2 + tid;
88
89 double sum = 0.0;
90 if (idx < size) sum += input[idx];
91 if (idx + blockDim.x < size) sum += input[idx + blockDim.x];
92
93 sdata[tid] = sum;
94 __syncthreads();
95
96 // Reduction in shared memory
97 for (unsigned int s = blockDim.x / 2; s > 0; s >>= 1) {
98 if (tid < s) sdata[tid] += sdata[tid + s];
99 __syncthreads();
100 }
101
102 if (tid == 0) output[blockIdx.x] = sdata[0];
103}
