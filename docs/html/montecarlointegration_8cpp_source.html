120static double monteCarloCPU(std::size_t num_samples,
121 std::size_t samples_per_iter) {
122 const std::size_t num_chunks =
123 (num_samples + samples_per_iter - 1) / samples_per_iter;
124
125 std::vector<std::size_t> chunks(num_chunks);
126 std::iota(chunks.begin(), chunks.end(), 0);
127
128 double sum = std::transform_reduce(
129 std::execution::par,
130 chunks.begin(), chunks.end(),
131 0.0,
132 std::plus<double>(),
133 [samples_per_iter](std::size_t) {
134 thread_local std::mt19937 rng(std::random_device{}());
135 thread_local std::uniform_real_distribution<double> dist(0.0, 1.0);
136
137 double local_sum = 0.0;
138 for (std::size_t i = 0; i < samples_per_iter; ++i) {
139 local_sum += f(dist(rng), dist(rng), dist(rng));
140 }
141 return local_sum;
142 }
143 );
144
145 return sum / static_cast<double>(num_samples);
146}

163static __global__ void monteCarloGPU(double* result,
164 std::size_t num_samples,
165 unsigned long long seed) {
166 const std::size_t idx =
167 blockIdx.x * blockDim.x + threadIdx.x;
168 const std::size_t stride =
169 gridDim.x * blockDim.x;
170
171 hiprandStatePhilox4_32_10_t state;
172 hiprand_init(seed, idx, 0, &state);
173
174 double local_sum = 0.0;
175 for (std::size_t i = idx; i < num_samples; i += stride) {
176 local_sum += f(
177 hiprand_uniform_double(&state),
178 hiprand_uniform_double(&state),
179 hiprand_uniform_double(&state)
180 );
181 }
182
183 atomicAdd(result, local_sum);
184}
