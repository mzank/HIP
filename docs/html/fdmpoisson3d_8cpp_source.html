195__host__ __device__ inline
196double rhsFunction(double x, double y, double z)
197{
198 const double sx = std::sin(x), sy = std::sin(y), sz = std::sin(z);
199 const double cx = std::cos(x), cy = std::cos(y), cz = std::cos(z);
200
201 const double g = sx * sy * sz;
202 const double gx = cx * sy * sz;
203 const double gy = sx * cy * sz;
204 const double gz = sx * sy * cz;
205 const double lap_g = -3.0 * g; // Î”g = -3g
206
207 const double xyz = x * y * z;
208 const double h = std::cos(xyz);
209 const double sh = std::sin(xyz);
210
211 const double hx = -y * z * sh;
212 const double hy = -x * z * sh;
213 const double hz = -x * y * sh;
214
215 const double hxx = -(y * y * z * z) * h;
216 const double hyy = -(x * x * z * z) * h;
217 const double hzz = -(x * x * y * y) * h;
218 const double lap_h = hxx + hyy + hzz;
219
220 const double grad_g_dot_grad_h = gx * hx + gy * hy + gz * hz;
221
222 return -(h * lap_g + g * lap_h + 2.0 * grad_g_dot_grad_h);
223}

257static void buildFDM3DPoisson(size_t Nx, size_t Ny, size_t Nz,
258 std::vector<int>& row_offset,
259 std::vector<int>& col,
260 std::vector<double>& val)
261{
262 const size_t Nx_i = Nx - 2;
263 const size_t Ny_i = Ny - 2;
264 const size_t Nz_i = Nz - 2;
265 const size_t N = Nx_i * Ny_i * Nz_i;
266
267 const double hx2 = 1.0 / std::pow(DOMAIN_LENGTH / static_cast<double>(Nx - 1), 2);
268 const double hy2 = 1.0 / std::pow(DOMAIN_LENGTH / static_cast<double>(Ny - 1), 2);
269 const double hz2 = 1.0 / std::pow(DOMAIN_LENGTH / static_cast<double>(Nz - 1), 2);
270
271 // Phase 1: compute nnz per row
272 std::vector<int> row_nnz(N);
273 for (size_t k = 0; k < Nz_i; ++k)
274 for (size_t j = 0; j < Ny_i; ++j)
275 for (size_t i = 0; i < Nx_i; ++i)
276 {
277 int nnz = 1;
278 if (i>0) ++nnz; if (i<Nx_i-1) ++nnz;
279 if (j>0) ++nnz; if (j<Ny_i-1) ++nnz;
280 if (k>0) ++nnz; if (k<Nz_i-1) ++nnz;
281 row_nnz[idx3D(i,j,k,Nx_i,Ny_i)] = nnz;
282 }
283
284 // Prefix sum -> row_offset
285 row_offset.resize(N + 1);
286 row_offset[0] = 0;
287 std::partial_sum(row_nnz.begin(), row_nnz.end(), row_offset.begin() + 1);
288
289 const size_t nnz_total = static_cast<size_t>(row_offset[N]);
290 col.resize(nnz_total);
291 val.resize(nnz_total);
292
293 // Phase 2: fill CSR entries in parallel
294 std::vector<size_t> rows(N);
295 std::iota(rows.begin(), rows.end(), 0);
296
297 std::for_each(std::execution::par, rows.begin(), rows.end(),
298 [&](size_t row)
299 {
300 const size_t k = row / (Nx_i * Ny_i);
301 const size_t j = (row / Nx_i) % Ny_i;
302 const size_t i = row % Nx_i;
303
304 size_t p = static_cast<size_t>(row_offset[row]);
305
306 col[p] = static_cast<int>(row);
307 val[p++] = 2.0 * (hx2 + hy2 + hz2);
308
309 if (i>0) { col[p] = static_cast<int>(idx3D(i-1,j,k,Nx_i,Ny_i)); val[p++] = -hx2; }
310 if (i<Nx_i-1) { col[p] = static_cast<int>(idx3D(i+1,j,k,Nx_i,Ny_i)); val[p++] = -hx2; }
311 if (j>0) { col[p] = static_cast<int>(idx3D(i,j-1,k,Nx_i,Ny_i)); val[p++] = -hy2; }
312 if (j<Ny_i-1) { col[p] = static_cast<int>(idx3D(i,j+1,k,Nx_i,Ny_i)); val[p++] = -hy2; }
313 if (k>0) { col[p] = static_cast<int>(idx3D(i,j,k-1,Nx_i,Ny_i)); val[p++] = -hz2; }
314 if (k<Nz_i-1) { col[p] = static_cast<int>(idx3D(i,j,k+1,Nx_i,Ny_i)); val[p++] = -hz2; }
315 });
316}

331template <typename Func>
332static void build3DVector(size_t Nx, size_t Ny, size_t Nz,
333 std::vector<double>& vec,
334 Func func)
335{
336 const size_t Nx_i = Nx - 2;
337 const size_t Ny_i = Ny - 2;
338 const size_t Nz_i = Nz - 2;
339 const size_t N = Nx_i * Ny_i * Nz_i;
340
341 const double hx = DOMAIN_LENGTH / static_cast<double>(Nx - 1);
342 const double hy = DOMAIN_LENGTH / static_cast<double>(Ny - 1);
343 const double hz = DOMAIN_LENGTH / static_cast<double>(Nz - 1);
344
345 vec.resize(N);
346 std::vector<size_t> indices(N);
347 std::iota(indices.begin(), indices.end(), 0);
348
349 std::transform(std::execution::par,
350 indices.begin(), indices.end(),
351 vec.begin(),
352 [&](size_t id)
353 {
354 const size_t i = id % Nx_i;
355 const size_t j = (id / Nx_i) % Ny_i;
356 const size_t k = id / (Nx_i * Ny_i);
357 return func(static_cast<double>(i+1)*hx,
358 static_cast<double>(j+1)*hy,
359 static_cast<double>(k+1)*hz);
360 });
361}

364// Compute L2 and Linf errors

373static std::pair<double,double> computeErrorL2Linf(const std::vector<double>& x,
374 const std::vector<double>& u_exact)
375{
376 const size_t N = x.size();
377
378 double l2 = std::sqrt(
379 std::transform_reduce(
380 std::execution::par,
381 x.begin(), x.end(),
382 u_exact.begin(),
383 0.0,
384 std::plus<>(),
385 [](double xi, double ui){ double e = xi - ui; return e*e; }
386 ) / static_cast<double>(N)
387 );
388
389 double linf = std::transform_reduce(
390 std::execution::par,
391 x.begin(), x.end(),
392 u_exact.begin(),
393 0.0,
394 [](double a, double b){ return std::max(a,b); },
395 [](double xi, double ui){ return std::abs(xi-ui); }
396 );
397
398 return {l2, linf};
399}
