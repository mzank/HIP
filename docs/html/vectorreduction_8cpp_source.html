81 static __global__ void block_reduce(const double* input,
82  double* output,
83  size_t size)
84 {
85  __shared__ double sdata[THREADS_PER_BLOCK];
86 
87  const unsigned int tid = threadIdx.x;
88  const size_t idx = static_cast<size_t>(blockIdx.x) * blockDim.x * 2 + tid;
89 
90  double sum = 0.0;
91  if (idx < size) sum += input[idx];
92  if (idx + blockDim.x < size) sum += input[idx + blockDim.x];
93 
94  sdata[tid] = sum;
95  __syncthreads();
96 
97  // Reduction in shared memory
98  for (unsigned int s = blockDim.x / 2; s > 0; s >>= 1) {
99  if (tid < s) sdata[tid] += sdata[tid + s];
100  __syncthreads();
101  }
102 
103  if (tid == 0) output[blockIdx.x] = sdata[0];
104 }
