244__host__ __device__ inline
245double exactSolution(double x, double y, double z)
246{
247 const double phi =
248 0.7 * std::cos(x)
249 + 0.5 * std::cos(2.0 * y)
250 + 0.3 * std::cos(3.0 * z)
251 + 0.2 * std::sin(x + y)
252 + 0.1 * std::sin(y + z)
253 + 0.05 * std::cos(28.0 * x)
254 + 0.05 * std::sin(27.0 * (y + z));
255
256 return std::exp(phi);
257}

262__host__ __device__ inline
263double rhsFunction(double x, double y, double z)
264{
265 const double phi =
266 0.7 * std::cos(x)
267 + 0.5 * std::cos(2.0 * y)
268 + 0.3 * std::cos(3.0 * z)
269 + 0.2 * std::sin(x + y)
270 + 0.1 * std::sin(y + z)
271 + 0.05 * std::cos(28.0 * x)
272 + 0.05 * std::sin(27.0 * (y + z));
273
274 const double phix =
275 -0.7 * std::sin(x)
276 + 0.2 * std::cos(x + y)
277 - 0.05 * 28.0 * std::sin(28.0 * x);
278
279 const double phiy =
280 -1.0 * std::sin(2.0 * y)
281 + 0.2 * std::cos(x + y)
282 + 0.1 * std::cos(y + z)
283 + 0.05 * 27.0 * std::cos(27.0 * (y + z));
284
285 const double phiz =
286 -0.9 * std::sin(3.0 * z)
287 + 0.1 * std::cos(y + z)
288 + 0.05 * 27.0 * std::cos(27.0 * (y + z));
289
290 const double phixx =
291 -0.7 * std::cos(x)
292 - 0.2 * std::sin(x + y)
293 - 0.05 * 28.0 * 28.0 * std::cos(28.0 * x);
294
295 const double phiyy =
296 -2.0 * std::cos(2.0 * y)
297 - 0.2 * std::sin(x + y)
298 - 0.1 * std::sin(y + z)
299 - 0.05 * 27.0 * 27.0 * std::sin(27.0 * (y + z));
300
301 const double phizz =
302 -2.7 * std::cos(3.0 * z)
303 - 0.1 * std::sin(y + z)
304 - 0.05 * 27.0 * 27.0 * std::sin(27.0 * (y + z));
305
306 const double lap_phi = phixx + phiyy + phizz;
307 const double grad_phi_sq = phix * phix + phiy * phiy + phiz * phiz;
308
309 return -(lap_phi + grad_phi_sq) * std::exp(phi);
310}

319static __global__
320void initRhsKernel(hipfftDoubleComplex* d_data,
321 size_t Nx, size_t Ny, size_t Nz)
322{
323 size_t i = blockIdx.x * blockDim.x + threadIdx.x;
324 size_t j = blockIdx.y * blockDim.y + threadIdx.y;
325 size_t k = blockIdx.z * blockDim.z + threadIdx.z;
326
327 if (i < Nx && j < Ny && k < Nz) {
328 size_t idx = (i * Ny + j) * Nz + k;
329
330 double x = L * static_cast<double>(i) / static_cast<double>(Nx);
331 double y = L * static_cast<double>(j) / static_cast<double>(Ny);
332 double z = L * static_cast<double>(k) / static_cast<double>(Nz);
333
334 d_data[idx].x = rhsFunction(x, y, z);
335 d_data[idx].y = 0.0;
336 }
337}

342static __global__
343void poissonFourierKernel(hipfftDoubleComplex* d_data,
344 size_t Nx, size_t Ny, size_t Nz)
345{
346 size_t i = blockIdx.x * blockDim.x + threadIdx.x;
347 size_t j = blockIdx.y * blockDim.y + threadIdx.y;
348 size_t k = blockIdx.z * blockDim.z + threadIdx.z;
349
350 if (i < Nx && j < Ny && k < Nz) {
351 size_t idx = (i * Ny + j) * Nz + k;
352
353 int ki = (i <= Nx / 2) ? int(i) : int(i) - int(Nx);
354 int kj = (j <= Ny / 2) ? int(j) : int(j) - int(Ny);
355 int kk = (k <= Nz / 2) ? int(k) : int(k) - int(Nz);
356
357 double k2 = double(ki * ki + kj * kj + kk * kk);
358
359 if (k2 > 0.0) {
360 d_data[idx].x /= k2;
361 d_data[idx].y /= k2;
362 } else {
363 d_data[idx].x = 0.0;
364 d_data[idx].y = 0.0;
365 }
366 }
367}

390static void poissonSolverGpu(size_t Nx, size_t Ny, size_t Nz,
391 hipfftHandle fftPlan,
392 hipfftDoubleComplex* d_solution)
393{
394 const size_t N = Nx * Ny * Nz;
395
396 dim3 threads(8, 8, 8);
397 dim3 blocks(static_cast<unsigned int>((Nx + threads.x - 1) / threads.x),
398 static_cast<unsigned int>((Ny + threads.y - 1) / threads.y),
399 static_cast<unsigned int>((Nz + threads.z - 1) / threads.z));
400
401 initRhsKernel<<<blocks, threads>>>(d_solution, Nx, Ny, Nz);
402 HIP_CHECK(hipDeviceSynchronize());
403
404 HIPFFT_CHECK(hipfftExecZ2Z(fftPlan, d_solution, d_solution, HIPFFT_FORWARD));
405 HIP_CHECK(hipDeviceSynchronize());
406
407 poissonFourierKernel<<<blocks, threads>>>(d_solution, Nx, Ny, Nz);
408 HIP_CHECK(hipDeviceSynchronize());
409
410 HIPFFT_CHECK(hipfftExecZ2Z(fftPlan, d_solution, d_solution, HIPFFT_BACKWARD));
411 HIP_CHECK(hipDeviceSynchronize());
412
413 const unsigned int threads1D = 256;
414 const unsigned int blocks1D = static_cast<unsigned int>((N + threads1D - 1) / threads1D);
415
416 normalizeKernel<<<blocks1D, threads1D>>>(d_solution, N, 1.0 / double(N));
417 HIP_CHECK(hipDeviceSynchronize());
418}

427static void poissonSolverCpu(size_t Nx, size_t Ny, size_t Nz,
428 fftw_complex* h_data)
429{
430 const size_t N = Nx * Ny * Nz;
431
432 std::vector<size_t> indices(N);
433 std::iota(indices.begin(), indices.end(), 0);
434
435 std::for_each(std::execution::par, indices.begin(), indices.end(),
436 [&](size_t idx) {
437 size_t i = idx / (Ny * Nz);
438 size_t j = (idx / Nz) % Ny;
439 size_t k = idx % Nz;
440
441 double x = L * static_cast<double>(i) / static_cast<double>(Nx);
442 double y = L * static_cast<double>(j) / static_cast<double>(Ny);
443 double z = L * static_cast<double>(k) / static_cast<double>(Nz);
444
445 h_data[idx][0] = rhsFunction(x, y, z);
446 h_data[idx][1] = 0.0;
447 });
448
449 fftw_plan forwardPlan =
450 fftw_plan_dft_3d(int(Nx), int(Ny), int(Nz),
451 h_data, h_data,
452 FFTW_FORWARD, FFTW_MEASURE);
453
454 fftw_plan backwardPlan =
455 fftw_plan_dft_3d(int(Nx), int(Ny), int(Nz),
456 h_data, h_data,
457 FFTW_BACKWARD, FFTW_MEASURE);
458
459 fftw_execute(forwardPlan);
460
461 std::for_each(std::execution::par, indices.begin(), indices.end(),
462 [&](size_t idx) {
463 size_t i = idx / (Ny * Nz);
464 size_t j = (idx / Nz) % Ny;
465 size_t k = idx % Nz;
466
467 int ki = (i <= Nx / 2) ? int(i) : int(i) - int(Nx);
468 int kj = (j <= Ny / 2) ? int(j) : int(j) - int(Ny);
469 int kk = (k <= Nz / 2) ? int(k) : int(k) - int(Nz);
470
471 double k2 = double(ki * ki + kj * kj + kk * kk);
472
473 if (k2 > 0.0) {
474 h_data[idx][0] /= k2;
475 h_data[idx][1] /= k2;
476 } else {
477 h_data[idx][0] = 0.0;
478 h_data[idx][1] = 0.0;
479 }
480 });
481
482 fftw_execute(backwardPlan);
483
484 const double scale = 1.0 / double(N);
485 std::for_each(std::execution::par, indices.begin(), indices.end(),
486 [&](size_t idx) {
487 h_data[idx][0] *= scale;
488 h_data[idx][1] *= scale;
489 });
490
491 fftw_destroy_plan(forwardPlan);
492 fftw_destroy_plan(backwardPlan);
493}
