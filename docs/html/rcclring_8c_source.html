107{
108 /* ------------------------- */
109 /* Initialize MPI */
110 /* ------------------------- */
111 MPI_Init(&argc, &argv);
112
113 int world_rank = 0, world_size = 0;
114 MPI_Comm_rank(MPI_COMM_WORLD, &world_rank);
115 MPI_Comm_size(MPI_COMM_WORLD, &world_size);
116
117 /* ------------------------- */
118 /* Optional NUMA and CPU affinity */
119 /* ------------------------- */
120#ifdef USE_NUMA
121 int cpu = sched_getcpu();
122 int node = numa_node_of_cpu(cpu);
123 numa_run_on_node(node); /* Bind process to NUMA node */
124 numa_set_localalloc(); /* Allocate future memory on local node */
125#endif

127 /* ------------------------- */
128 /* HIP device selection */
129 /* ------------------------- */
130 int num_devices = 0;
131 HIP_CHECK(hipGetDeviceCount(&num_devices));
132 HIP_CHECK(hipSetDevice(world_rank % num_devices));
133
134 /* ------------------------- */
135 /* RCCL initialization */
136 /* ------------------------- */
137 ncclUniqueId id;
138 if (world_rank == 0)
139 RCCL_CHECK(ncclGetUniqueId(&id));
140
141 MPI_Bcast(&id, sizeof(id), MPI_BYTE, 0, MPI_COMM_WORLD);
142
143 ncclComm_t comm;
144 RCCL_CHECK(ncclCommInitRank(&comm, world_size, id, world_rank));

147 HIP_CHECK(hipStreamCreate(&stream));
148
149 const int next = (world_rank + 1) % world_size;
150 const int prev = (world_rank - 1 + world_size) % world_size;
151
152 /* ------------------------- */
153 /* Print header */
154 /* ------------------------- */
155 if (world_rank == 0) {
156 printf("\nMsg size (MB) |");
157 for (int r = 0; r < world_size; r++)
158 printf(" Rank %d BW (GB/s) | Send[0] | Recv[0] |", r);
159 printf("\n");
160 }

162 /* ------------------------- */
163 /* Loop over message sizes */
164 /* ------------------------- */
165 for (size_t msg_size = MIN_MSG_SIZE;
166 msg_size <= MAX_MSG_SIZE;
167 msg_size <<= 1)
168 {
169 const size_t count = msg_size / sizeof(double);
170 if (count > INT_MAX) {
171 fprintf(stderr, "Message too large for MPI count (%zu elements)\n", count);
172 MPI_Abort(MPI_COMM_WORLD, -1);
173 }

175 /* ------------------------- */
176 /* Allocate GPU buffers */
177 /* ------------------------- */
178 double *d_send = NULL;
179 double *d_recv = NULL;

181 HIP_CHECK(hipMalloc((void**)&d_recv, msg_size));

183 /* ------------------------- */

185 /* ------------------------- */

189 h_init[i] = (double)(world_rank + 1);

191 HIP_CHECK(hipMemcpy(d_send, h_init, msg_size, hipMemcpyHostToDevice));
192 free(h_init);

194 HIP_CHECK(hipDeviceSynchronize());

196 /* ------------------------- */
197 /* Warm-up iterations */
198 /* ------------------------- */
199 for (int i = 0; i < N_WARMUP; i++) {
200 RCCL_CHECK(ncclGroupStart());

202 RCCL_CHECK(ncclSend(d_send, count, ncclDouble, next, comm, stream));
203 RCCL_CHECK(ncclGroupEnd());
204 }

207 /* ------------------------- */
208 /* Timed iterations */
209 /* ------------------------- */

211 hipEvent_t start, stop;

213 HIP_CHECK(hipEventCreate(&stop));
214
215 for (int rep = 0; rep < N_REPEAT; rep++) {
216 HIP_CHECK(hipEventRecord(start, stream));
217
218 RCCL_CHECK(ncclGroupStart());
219 RCCL_CHECK(ncclRecv(d_recv, count, ncclDouble, prev, comm, stream));
220 RCCL_CHECK(ncclSend(d_send, count, ncclDouble, next, comm, stream));
221 RCCL_CHECK(ncclGroupEnd());
222
223 HIP_CHECK(hipEventRecord(stop, stream));
224 HIP_CHECK(hipEventSynchronize(stop));

229 }

231 HIP_CHECK(hipDeviceSynchronize());
232
233 /* ------------------------- */
234 /* Verification of first element */
235 /* ------------------------- */
236 double send0 = 0.0, recv0 = 0.0;
237 HIP_CHECK(hipMemcpy(&send0, d_send, sizeof(double), hipMemcpyDeviceToHost));
238 HIP_CHECK(hipMemcpy(&recv0, d_recv, sizeof(double), hipMemcpyDeviceToHost));
239
240 /* ------------------------- */
241 /* Compute bandwidth (GB/s) */
242 /* ------------------------- */
243 const double avg_s = ((double)total_ms / N_REPEAT) * 1.0e-3;
244 const double bw_GBps = (2.0 * (double)msg_size / avg_s) * 1.0e-9;
245
246 /* ------------------------- */
247 /* Gather results to rank 0 */
248 /* ------------------------- */
249 double *all_bw = NULL;
250 double *all_send0 = NULL;
251 double *all_recv0 = NULL;
252
253 if (world_rank == 0) {
254 const size_t n = (size_t)world_size;
255 all_bw = malloc(n * sizeof(double));
256 all_send0 = malloc(n * sizeof(double));
257 all_recv0 = malloc(n * sizeof(double));
258 CHECK_ALLOC(all_bw);
259 CHECK_ALLOC(all_send0);
260 CHECK_ALLOC(all_recv0);
261 }
262
263 MPI_Gather(&bw_GBps, 1, MPI_DOUBLE, all_bw, 1, MPI_DOUBLE, 0, MPI_COMM_WORLD);
264 MPI_Gather(&send0, 1, MPI_DOUBLE, all_send0, 1, MPI_DOUBLE, 0, MPI_COMM_WORLD);
265 MPI_Gather(&recv0, 1, MPI_DOUBLE, all_recv0, 1, MPI_DOUBLE, 0, MPI_COMM_WORLD);
266
267 /* ------------------------- */

269 /* ------------------------- */
270 if (world_rank == 0) {
271 printf("%13.2f |", (double)msg_size * 1.0e-6);
272 for (int r = 0; r < world_size; r++) {
273 printf(" %16.2f | %7.2f | %7.2f |",
274 all_bw[r], all_send0[r], all_recv0[r]);
275 }
276 printf("\n");
277
278 free(all_bw);
279 free(all_send0);
280 free(all_recv0);
281 }
282
283 /* ------------------------- */
284 /* Cleanup buffers */
285 /* ------------------------- */
286 HIP_CHECK(hipFree(d_send));
287 HIP_CHECK(hipFree(d_recv));
288 HIP_CHECK(hipEventDestroy(start));
289 HIP_CHECK(hipEventDestroy(stop));
290 }
291
292 /* ------------------------- */
293 /* Cleanup RCCL and HIP */
294 /* ------------------------- */
295 RCCL_CHECK(ncclCommDestroy(comm));
296 HIP_CHECK(hipStreamDestroy(stream));
297
298 MPI_Finalize();
299 return EXIT_SUCCESS;
300}
