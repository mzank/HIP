122__host__ __device__ inline
123double exactSolution(double x, double y, double z)
124{
125 const double phi =
126 0.7 * std::cos(x)
127 + 0.5 * std::cos(2.0 * y)
128 + 0.3 * std::cos(3.0 * z)
129 + 0.2 * std::sin(x + y)
130 + 0.1 * std::sin(y + z)
131 + 0.05 * std::cos(28.0 * x)
132 + 0.05 * std::sin(27.0 * (y + z));
133
134 return std::exp(phi);
135}

140__host__ __device__ inline
141double rhsFunction(double x, double y, double z)
142{
143 const double phi =
144 0.7 * std::cos(x)
145 + 0.5 * std::cos(2.0 * y)
146 + 0.3 * std::cos(3.0 * z)
147 + 0.2 * std::sin(x + y)
148 + 0.1 * std::sin(y + z)
149 + 0.05 * std::cos(28.0 * x)
150 + 0.05 * std::sin(27.0 * (y + z));
151
152 const double phix =
153 -0.7 * std::sin(x)
154 + 0.2 * std::cos(x + y)
155 - 0.05 * 28.0 * std::sin(28.0 * x);
156
157 const double phiy =
158 -1.0 * std::sin(2.0 * y)
159 + 0.2 * std::cos(x + y)
160 + 0.1 * std::cos(y + z)
161 + 0.05 * 27.0 * std::cos(27.0 * (y + z));
162
163 const double phiz =
164 -0.9 * std::sin(3.0 * z)
165 + 0.1 * std::cos(y + z)
166 + 0.05 * 27.0 * std::cos(27.0 * (y + z));
167
168 const double phixx =
169 -0.7 * std::cos(x)
170 - 0.2 * std::sin(x + y)
171 - 0.05 * 28.0 * 28.0 * std::cos(28.0 * x);
172
173 const double phiyy =
174 -2.0 * std::cos(2.0 * y)
175 - 0.2 * std::sin(x + y)
176 - 0.1 * std::sin(y + z)
177 - 0.05 * 27.0 * 27.0 * std::sin(27.0 * (y + z));
178
179 const double phizz =
180 -2.7 * std::cos(3.0 * z)
181 - 0.1 * std::sin(y + z)
182 - 0.05 * 27.0 * 27.0 * std::sin(27.0 * (y + z));
183
184 const double lap_phi = phixx + phiyy + phizz;
185 const double grad_phi_sq = phix * phix + phiy * phiy + phiz * phiz;
186
187 return -(lap_phi + grad_phi_sq) * std::exp(phi);
188}

197static __global__
198void initRhsKernel(hipfftDoubleComplex* d_data,
199 size_t Nx, size_t Ny, size_t Nz)
200{
201 size_t i = blockIdx.x * blockDim.x + threadIdx.x;
202 size_t j = blockIdx.y * blockDim.y + threadIdx.y;
203 size_t k = blockIdx.z * blockDim.z + threadIdx.z;
204
205 if (i < Nx && j < Ny && k < Nz) {
206 size_t idx = (i * Ny + j) * Nz + k;
207
208 double x = L * static_cast<double>(i) / static_cast<double>(Nx);
209 double y = L * static_cast<double>(j) / static_cast<double>(Ny);
210 double z = L * static_cast<double>(k) / static_cast<double>(Nz);
211
212 d_data[idx].x = rhsFunction(x, y, z);
213 d_data[idx].y = 0.0;
214 }
215}

220static __global__
221void poissonFourierKernel(hipfftDoubleComplex* d_data,
222 size_t Nx, size_t Ny, size_t Nz)
223{
224 size_t i = blockIdx.x * blockDim.x + threadIdx.x;
225 size_t j = blockIdx.y * blockDim.y + threadIdx.y;
226 size_t k = blockIdx.z * blockDim.z + threadIdx.z;
227
228 if (i < Nx && j < Ny && k < Nz) {
229 size_t idx = (i * Ny + j) * Nz + k;
230
231 int ki = (i <= Nx / 2) ? int(i) : int(i) - int(Nx);
232 int kj = (j <= Ny / 2) ? int(j) : int(j) - int(Ny);
233 int kk = (k <= Nz / 2) ? int(k) : int(k) - int(Nz);
234
235 double k2 = double(ki * ki + kj * kj + kk * kk);
236
237 if (k2 > 0.0) {
238 d_data[idx].x /= k2;
239 d_data[idx].y /= k2;
240 } else {
241 d_data[idx].x = 0.0;
242 d_data[idx].y = 0.0;
243 }
244 }
245}

268static void poissonSolverGpu(size_t Nx, size_t Ny, size_t Nz,
269 hipfftHandle fftPlan,
270 hipfftDoubleComplex* d_solution)
271{
272 const size_t N = Nx * Ny * Nz;
273
274 dim3 threads(8, 8, 8);
275 dim3 blocks(static_cast<unsigned int>((Nx + threads.x - 1) / threads.x),
276 static_cast<unsigned int>((Ny + threads.y - 1) / threads.y),
277 static_cast<unsigned int>((Nz + threads.z - 1) / threads.z));
278
279 initRhsKernel<<<blocks, threads>>>(d_solution, Nx, Ny, Nz);
280 HIP_CHECK(hipDeviceSynchronize());
281
282 HIPFFT_CHECK(hipfftExecZ2Z(fftPlan, d_solution, d_solution, HIPFFT_FORWARD));
283 HIP_CHECK(hipDeviceSynchronize());
284
285 poissonFourierKernel<<<blocks, threads>>>(d_solution, Nx, Ny, Nz);
286 HIP_CHECK(hipDeviceSynchronize());
287
288 HIPFFT_CHECK(hipfftExecZ2Z(fftPlan, d_solution, d_solution, HIPFFT_BACKWARD));
289 HIP_CHECK(hipDeviceSynchronize());
290
291 const unsigned int threads1D = 256;
292 const unsigned int blocks1D = static_cast<unsigned int>((N + threads1D - 1) / threads1D);
293
294 normalizeKernel<<<blocks1D, threads1D>>>(d_solution, N, 1.0 / double(N));
295 HIP_CHECK(hipDeviceSynchronize());
296}

305static void poissonSolverCpu(size_t Nx, size_t Ny, size_t Nz,
306 fftw_complex* h_data)
307{
308 const size_t N = Nx * Ny * Nz;
309
310 std::vector<size_t> indices(N);
311 std::iota(indices.begin(), indices.end(), 0);
312
313 std::for_each(std::execution::par, indices.begin(), indices.end(),
314 [&](size_t idx) {
315 size_t i = idx / (Ny * Nz);
316 size_t j = (idx / Nz) % Ny;
317 size_t k = idx % Nz;
318
319 double x = L * static_cast<double>(i) / static_cast<double>(Nx);
320 double y = L * static_cast<double>(j) / static_cast<double>(Ny);
321 double z = L * static_cast<double>(k) / static_cast<double>(Nz);
322
323 h_data[idx][0] = rhsFunction(x, y, z);
324 h_data[idx][1] = 0.0;
325 });
326
327 fftw_plan forwardPlan =
328 fftw_plan_dft_3d(int(Nx), int(Ny), int(Nz),
329 h_data, h_data,
330 FFTW_FORWARD, FFTW_MEASURE);
331
332 fftw_plan backwardPlan =
333 fftw_plan_dft_3d(int(Nx), int(Ny), int(Nz),
334 h_data, h_data,
335 FFTW_BACKWARD, FFTW_MEASURE);
336
337 fftw_execute(forwardPlan);
338
339 std::for_each(std::execution::par, indices.begin(), indices.end(),
340 [&](size_t idx) {
341 size_t i = idx / (Ny * Nz);
342 size_t j = (idx / Nz) % Ny;
343 size_t k = idx % Nz;
344
345 int ki = (i <= Nx / 2) ? int(i) : int(i) - int(Nx);
346 int kj = (j <= Ny / 2) ? int(j) : int(j) - int(Ny);
347 int kk = (k <= Nz / 2) ? int(k) : int(k) - int(Nz);
348
349 double k2 = double(ki * ki + kj * kj + kk * kk);
350
351 if (k2 > 0.0) {
352 h_data[idx][0] /= k2;
353 h_data[idx][1] /= k2;
354 } else {
355 h_data[idx][0] = 0.0;
356 h_data[idx][1] = 0.0;
357 }
358 });
359
360 fftw_execute(backwardPlan);
361
362 const double scale = 1.0 / double(N);
363 std::for_each(std::execution::par, indices.begin(), indices.end(),
364 [&](size_t idx) {
365 h_data[idx][0] *= scale;
366 h_data[idx][1] *= scale;
367 });
368
369 fftw_destroy_plan(forwardPlan);
370 fftw_destroy_plan(backwardPlan);
371}
