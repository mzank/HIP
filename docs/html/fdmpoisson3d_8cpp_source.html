105__host__ __device__ inline
106double rhsFunction(double x, double y, double z)
107{
108 const double sx = std::sin(x), sy = std::sin(y), sz = std::sin(z);
109 const double cx = std::cos(x), cy = std::cos(y), cz = std::cos(z);
110
111 const double g = sx * sy * sz;
112 const double gx = cx * sy * sz;
113 const double gy = sx * cy * sz;
114 const double gz = sx * sy * cz;
115 const double lap_g = -3.0 * g; // Î”g = -3g
116
117 const double xyz = x * y * z;
118 const double h = std::cos(xyz);
119 const double sh = std::sin(xyz);
120
121 const double hx = -y * z * sh;
122 const double hy = -x * z * sh;
123 const double hz = -x * y * sh;
124
125 const double hxx = -(y * y * z * z) * h;
126 const double hyy = -(x * x * z * z) * h;
127 const double hzz = -(x * x * y * y) * h;
128 const double lap_h = hxx + hyy + hzz;
129
130 const double grad_g_dot_grad_h = gx * hx + gy * hy + gz * hz;
131
132 return -(h * lap_g + g * lap_h + 2.0 * grad_g_dot_grad_h);
133}

167static void buildFDM3DPoisson(size_t Nx, size_t Ny, size_t Nz,
168 std::vector<int>& row_offset,
169 std::vector<int>& col,
170 std::vector<double>& val)
171{
172 const size_t Nx_i = Nx - 2;
173 const size_t Ny_i = Ny - 2;
174 const size_t Nz_i = Nz - 2;
175 const size_t N = Nx_i * Ny_i * Nz_i;
176
177 const double hx2 = 1.0 / std::pow(DOMAIN_LENGTH / static_cast<double>(Nx - 1), 2);
178 const double hy2 = 1.0 / std::pow(DOMAIN_LENGTH / static_cast<double>(Ny - 1), 2);
179 const double hz2 = 1.0 / std::pow(DOMAIN_LENGTH / static_cast<double>(Nz - 1), 2);
180
181 // Phase 1: compute nnz per row
182 std::vector<int> row_nnz(N);
183 for (size_t k = 0; k < Nz_i; ++k)
184 for (size_t j = 0; j < Ny_i; ++j)
185 for (size_t i = 0; i < Nx_i; ++i)
186 {
187 int nnz = 1;
188 if (i>0) ++nnz; if (i<Nx_i-1) ++nnz;
189 if (j>0) ++nnz; if (j<Ny_i-1) ++nnz;
190 if (k>0) ++nnz; if (k<Nz_i-1) ++nnz;
191 row_nnz[idx3D(i,j,k,Nx_i,Ny_i)] = nnz;
192 }
193
194 // Prefix sum -> row_offset
195 row_offset.resize(N + 1);
196 row_offset[0] = 0;
197 std::partial_sum(row_nnz.begin(), row_nnz.end(), row_offset.begin() + 1);
198
199 const size_t nnz_total = static_cast<size_t>(row_offset[N]);
200 col.resize(nnz_total);
201 val.resize(nnz_total);
202
203 // Phase 2: fill CSR entries in parallel
204 std::vector<size_t> rows(N);
205 std::iota(rows.begin(), rows.end(), 0);
206
207 std::for_each(std::execution::par, rows.begin(), rows.end(),
208 [&](size_t row)
209 {
210 const size_t k = row / (Nx_i * Ny_i);
211 const size_t j = (row / Nx_i) % Ny_i;
212 const size_t i = row % Nx_i;
213
214 size_t p = static_cast<size_t>(row_offset[row]);
215
216 col[p] = static_cast<int>(row);
217 val[p++] = 2.0 * (hx2 + hy2 + hz2);
218
219 if (i>0) { col[p] = static_cast<int>(idx3D(i-1,j,k,Nx_i,Ny_i)); val[p++] = -hx2; }
220 if (i<Nx_i-1) { col[p] = static_cast<int>(idx3D(i+1,j,k,Nx_i,Ny_i)); val[p++] = -hx2; }
221 if (j>0) { col[p] = static_cast<int>(idx3D(i,j-1,k,Nx_i,Ny_i)); val[p++] = -hy2; }
222 if (j<Ny_i-1) { col[p] = static_cast<int>(idx3D(i,j+1,k,Nx_i,Ny_i)); val[p++] = -hy2; }
223 if (k>0) { col[p] = static_cast<int>(idx3D(i,j,k-1,Nx_i,Ny_i)); val[p++] = -hz2; }
224 if (k<Nz_i-1) { col[p] = static_cast<int>(idx3D(i,j,k+1,Nx_i,Ny_i)); val[p++] = -hz2; }
225 });
226}

241template <typename Func>
242static void build3DVector(size_t Nx, size_t Ny, size_t Nz,
243 std::vector<double>& vec,
244 Func func)
245{
246 const size_t Nx_i = Nx - 2;
247 const size_t Ny_i = Ny - 2;
248 const size_t Nz_i = Nz - 2;
249 const size_t N = Nx_i * Ny_i * Nz_i;
250
251 const double hx = DOMAIN_LENGTH / static_cast<double>(Nx - 1);
252 const double hy = DOMAIN_LENGTH / static_cast<double>(Ny - 1);
253 const double hz = DOMAIN_LENGTH / static_cast<double>(Nz - 1);
254
255 vec.resize(N);
256 std::vector<size_t> indices(N);
257 std::iota(indices.begin(), indices.end(), 0);
258
259 std::transform(std::execution::par,
260 indices.begin(), indices.end(),
261 vec.begin(),
262 [&](size_t id)
263 {
264 const size_t i = id % Nx_i;
265 const size_t j = (id / Nx_i) % Ny_i;
266 const size_t k = id / (Nx_i * Ny_i);
267 return func(static_cast<double>(i+1)*hx,
268 static_cast<double>(j+1)*hy,
269 static_cast<double>(k+1)*hz);
270 });
271}

283static std::pair<double,double> computeErrorL2Linf(const std::vector<double>& x,
284 const std::vector<double>& u_exact)
285{
286 const size_t N = x.size();
287
288 double l2 = std::sqrt(
289 std::transform_reduce(
290 std::execution::par,
291 x.begin(), x.end(),
292 u_exact.begin(),
293 0.0,
294 std::plus<>(),
295 [](double xi, double ui){ double e = xi - ui; return e*e; }
296 ) / static_cast<double>(N)
297 );
298
299 double linf = std::transform_reduce(
300 std::execution::par,
301 x.begin(), x.end(),
302 u_exact.begin(),
303 0.0,
304 [](double a, double b){ return std::max(a,b); },
305 [](double xi, double ui){ return std::abs(xi-ui); }
306 );
307
308 return {l2, linf};
309}
